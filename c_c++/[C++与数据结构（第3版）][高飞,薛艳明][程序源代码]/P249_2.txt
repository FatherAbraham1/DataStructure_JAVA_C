void Graph < T,VerType >::DFS()
  {	 int n;
	   	 n = GetNumVex();       						//取顶点个数
	  	 visited = new int[n];   					//根据顶点个数创建辅助数组
	 	 for(int i = 0;i < n;i++)visited[i] = 0; 	//辅助数组初始化
	  	 DFS(0);										//从顶点0开始深度优先搜索
	 	 delete []visited;
  }

 	 while(w!= -1)       							//有邻接顶点
 	   {  if(!visited[w])DFS(w);   				//若未访问过，从w递归访问

  }
